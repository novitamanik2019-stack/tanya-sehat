<script>
document.getElementById("questionForm").addEventListener("submit", function(e) {
e.preventDefault();

let nama = document.getElementById("nama").value;
let pertanyaan = document.getElementById("pertanyaan").value;

let questionBox = document.createElement("div");
questionBox.classList.add("question-box");

questionBox.innerHTML = `
<h4>${nama} bertanya:</h4>
<p>${pertanyaan}</p>
<button class="jawabBtn">Jawab</button>
<div class="answer-section"></div>
`;

document.getElementById("questionList").appendChild(questionBox);

// reset form
document.getElementById("questionForm").reset();

// Tambahkan event untuk tombol Jawab
let jawabBtn = questionBox.querySelector(".jawabBtn");
jawabBtn.addEventListener("click", function() {
let answerSection = questionBox.querySelector(".answer-section");

// kalau form sudah ada, jangan ditambah lagi
if (answerSection.querySelector("form")) return;

let form = document.createElement("form");
form.classList.add("answer-form");

form.innerHTML = `
<textarea placeholder="Tulis jawaban di sini..." required></textarea>
<button type="submit">Kirim Jawaban</button>
`;

form.addEventListener("submit", function(e) {
e.preventDefault();
let jawaban = form.querySelector("textarea").value;

let answerDiv = document.createElement("div");
answerDiv.classList.add("answer");
answerDiv.innerHTML = `<b>Jawaban:</b> <p>${jawaban}</p>`;

answerSection.innerHTML = ""; // hapus form
answerSection.appendChild(answerDiv);
});

answerSection.appendChild(form);
});
});
</script>
